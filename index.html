    <script>

        class MedicalAppointmentApp {

            constructor() {

                this.appointments = JSON.parse(localStorage.getItem('appointments') || '[]');

                this.mediaRecorder = null;

                this.audioChunks = [];

                this.currentAudioBlob = null;

                this.isRecording = false;

                

                this.initializeApp();

            }

 

            initializeApp() {

                this.renderAppointments();

                this.setupEventListeners();

                this.setMinDate();

            }

 

            setMinDate() {

                const today = new Date().toISOString().split('T')[0];

                document.getElementById('date').min = today;

            }

 

            setupEventListeners() {

                document.getElementById('appointmentForm').addEventListener('submit', (e) => {

                    e.preventDefault();

                    this.addAppointment();

                });

 

                document.getElementById('recordBtn').addEventListener('click', () => {

                    this.toggleRecording();

                });

 

                document.getElementById('playBtn').addEventListener('click', () => {

                    this.playRecording();

                });

            }

 

            async toggleRecording() {

                const recordBtn = document.getElementById('recordBtn');

                const voiceStatus = document.getElementById('voiceStatus');

 

                if (!this.isRecording) {

                    try {

                        const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

                        this.mediaRecorder = new MediaRecorder(stream);

                        this.audioChunks = [];

 

                        this.mediaRecorder.ondataavailable = (event) => {

                            this.audioChunks.push(event.data);

                        };

 

                        this.mediaRecorder.onstop = () => {

                            this.currentAudioBlob = new Blob(this.audioChunks, { type: 'audio/wav' });

                            document.getElementById('playBtn').style.display = 'inline-flex';

                            voiceStatus.textContent = 'Nota de voz guardada â';

                            voiceStatus.style.color = '#27ae60';

                        };

 

                        this.mediaRecorder.start();

                        this.isRecording = true;

                        recordBtn.textContent = 'â¹ï¸ Detener';

                        recordBtn.classList.add('recording');

                        voiceStatus.textContent = 'Grabando...';

                        voiceStatus.className = 'status-indicator recording-indicator';

 

                    } catch (error) {

                        alert('Error al acceder al micrÃ³fono. AsegÃºrate de dar permisos.');

                        console.error('Error:', error);

                    }

                } else {

                    this.mediaRecorder.stop();

                    this.mediaRecorder.stream.getTracks().forEach(track => track.stop());

                    this.isRecording = false;

                    recordBtn.textContent = '🎤 Grabar nota';

                    recordBtn.classList.remove('recording');

                }

            }

 

            playRecording() {

                if (this.currentAudioBlob) {

                    const audio = new Audio(URL.createObjectURL(this.currentAudioBlob));

                    audio.play();

                }

            }

 

            addAppointment() {

                const formData = new FormData(document.getElementById('appointmentForm'));

                const appointment = {

                    id: Date.now(),

                    doctor: formData.get('doctor'),

                    date: formData.get('date'),

                    time: formData.get('time'),

                    location: formData.get('location'),

                    notes: formData.get('notes'),

                    voiceNote: this.currentAudioBlob ? URL.createObjectURL(this.currentAudioBlob) : null,

                    created: new Date().toISOString()

                };

 

                this.appointments.push(appointment);

                this.saveAppointments();

                this.renderAppointments();

                this.resetForm();

            }

 

            saveAppointments() {

                // Note: We can't store audio blobs in localStorage, so we'll just save the text data

                const appointmentsToSave = this.appointments.map(apt => ({

                    ...apt,

                    voiceNote: apt.voiceNote ? 'audio_recorded' : null

                }));

                localStorage.setItem('appointments', JSON.stringify(appointmentsToSave));

            }

 

            renderAppointments() {

                const container = document.getElementById('appointmentsList');

                

                if (this.appointments.length === 0) {

                    container.innerHTML = '<div class="empty-state">No tienes citas programadas</div>';

                    return;

                }

 

                const sortedAppointments = this.appointments.sort((a, b) => {

                    return new Date(a.date + ' ' + a.time) - new Date(b.date + ' ' + b.time);

                });

 

                container.innerHTML = sortedAppointments.map(appointment => `

                    <div class="appointment-card">

                        <div class="appointment-header">

                            <div class="appointment-title">${appointment.doctor}</div>

                            <div class="appointment-date">${this.formatDate(appointment.date)} ${appointment.time}</div>

                        </div>

                        <div class="appointment-details">

                            ${appointment.location ? `📍 ${appointment.location}` : ''}

                            ${appointment.notes ? `<br>📝 ${appointment.notes}` : ''}

                        </div>

                        <div class="voice-note-controls">

                            ${appointment.voiceNote ? `

                                <button class="voice-btn play-btn" onclick="app.playAppointmentVoice('${appointment.id}')">

                                    â¶ï¸ Nota de voz

                                </button>

                            ` : ''}

                            <button class="delete-btn" onclick="app.deleteAppointment(${appointment.id})">

                                🗑ï¸ Eliminar

                            </button>

                        </div>

                    </div>

                `).join('');

            }

 

            formatDate(dateString) {

                const options = { 

                    weekday: 'short', 

                    year: 'numeric', 

                    month: 'short', 

                    day: 'numeric' 

                };

                return new Date(dateString).toLocaleDateString('es-ES', options);

            }

 

            deleteAppointment(id) {

                if (confirm('Â¿EstÃ¡s seguro de que quieres eliminar esta cita?')) {

                    this.appointments = this.appointments.filter(apt => apt.id !== id);

                    this.saveAppointments();

                    this.renderAppointments();

                }

            }

 

            playAppointmentVoice(id) {

                const appointment = this.appointments.find(apt => apt.id == id);

                if (appointment && appointment.voiceNote) {

                    const audio = new Audio(appointment.voiceNote);

                    audio.play();

                }

            }

 

            resetForm() {

                document.getElementById('appointmentForm').reset();

                document.getElementById('playBtn').style.display = 'none';

                document.getElementById('voiceStatus').textContent = '';

                this.currentAudioBlob = null;

                this.setMinDate();

            }

        }

 

        // Initialize the app

        const app = new MedicalAppointmentApp();

 

        // Service Worker for offline functionality (basic)

        if ('serviceWorker' in navigator) {

            navigator.serviceWorker.register('data:text/javascript,')

                .catch(err => console.log('SW registration failed'));

        }

    </script>
